21 ## @fn setup_network()
22 ## @brief Call net-setup and dhcpcd to enable networking
23 ## @details Create file \b setup_network on success for debugging purposes
24 ## @retval Otherwise exit -1 on failure
25 ## @ingroup mkFileSystem
26 
27 setup_network() {
28  if test -f setup_network; then
29  return
30  fi
31  if test "${VMTYPE}" = "gui"; then
32  dhcpcd -HD $(ifconfig | cut -f1 -d' ' | line | cut -f1 -d':')
33  else
34  net-setup
35  fi
36  if test $? = 0; then
37  touch setup_network
38  else
39  echo "Could not fix internet access" | tee setup_network.log
40  sleep 10
41  exit -1
42  fi
43 }

45 ## @fn partition()
46 ## @brief Create partition table, \b /dev/sda1 (bios_grub), \b /dev/sda2 (boot), \b dev/sda3 (swap) and \b /dev/sda4 (system)
47 ## @details Create file \b partition. @n
48 ## On error, fill this file with successive exit codes of commands and exit.@n
49 ## On success, just create empty file.
50 ## @ingroup mkFileSystem
51 
52 partition() {
53  if test -f partition; then
54  return
55  fi
56  parted --script --align=opt /dev/sda "mklabel gpt unit mib mkpart primary 1 3 name 1 grub set 1 bios_grub on mkpart primary 3 131 name 2 boot mkpart primary 131 643 name 3 swap mkpart primary 643 -1 set 2 boot on"
57  res0=$?
58  mkfs.fat -F 32 /dev/sda2
59  res1=$?
60  mkfs.ext4 /dev/sda4
61  res2=$?
62  mkswap /dev/sda3
63  res3=$?
64  swapon /dev/sda3
65  res4=$?
66  mount /dev/sda4 /mnt/gentoo
67  res5=$?
68  res=$((${res0} | ${res1} | ${res2} | ${res3} | ${res4} | ${res5}))
69  if test ${res} = 0; then
70  touch partition
71  else
72  echo "parted exit code: ${res0}" > partition
73  echo "mkfs.fat exit code: ${res1}" >> partition
74  echo "mkfs.ext4 exit code: ${res2}" >> partition
75  echo "mkswap exit code: ${res3}" >> partition
76  echo "swapon exit code: ${res4}" >> partition
77  echo "mount exit code: ${res5}" >> partition
78  echo "Failed to partition main disk"
79  sleep 10
80  exit -1
81  fi
82 }

84 ## @fn install_stage3()
85 ## @details @li Copy stage3 archive, ebuild list, kernel config and mkvm_chroot.sh to /mnt/gentoo
86 ## @li Extract it, fix basic make.conf options
87 ## @li Install repos.conf from liveCD (networking parameters)
88 ## @li Mount liveCD proc/sys/dev files into new filesystem
89 ## @li On failure, print error codes into file \b partition
90 ## @li On success, create empty file \b partition
91 ## @li <tt> chroot </tt> into new system
92 ## @ingroup mkFileSystem
93 
94 install_stage3() {
95  if test -f stage3; then
96  return
97  fi
98  mv -vf ${STAGE3} ${ELIST} mkvm_chroot.sh ${KERNEL_CONFIG} /mnt/gentoo/
99  cp -vf .bashrc /mnt/gentoo/bashrc_temp
100  cd /mnt/gentoo
101  head -n -1 -q bashrc_temp > temp_bashrc && rm bashrc_temp
102 
103  # Time
104 
105  ntpd -q -g
106  tar xpJf ${STAGE3} --xattrs-include='*.*' --numeric-owner
107  if test $? != 0; then
108  echo "stage3 tarball could not be extracted"
109  sleep 10
110  exit -1
111  fi
112  cat temp_bashrc >> .bashrc
113  rm temp_bashrc
114  rm -vf ${STAGE3}
115 
116  # Ajusting portage
117 
118  local m_conf="etc/portage/make.conf"
119  sed -i "s/COMMON_FLAGS=.*/COMMON_FLAGS='${CFLAGS} -pipe'/g" ${m_conf}
120  echo MAKEOPTS=-j${NCPUS} >> ${m_conf}
121  echo 'L10N="fr en"' >> ${m_conf}
122  echo 'LINGUAS="fr en"' >> ${m_conf}
123  sed -i 's/USE=".*"//g' ${m_conf}
124  echo 'USE="-gtk -gnome qt4 qt5 kde dvd alsa cdr bindist networkmanager elogind -consolekit -systemd dbus X"' >> ${m_conf}
125  echo "GENTOO_MIRRORS=${EMIRRORS}" >> ${m_conf}
126 
127  # note linux-fw-redistributable no-source-code for genkernel
128  # note bh-luxi for font-bh-* requested by xorg-x11
129 
130  echo 'ACCEPT_LICENSE="-* @FREE linux-fw-redistributable no-source-code bh-luxi"' >> ${m_conf}
131  echo 'GRUB_PLATFORMS="efi-64"' >> ${m_conf}
132  echo 'VIDEO_CARDS="nouveau intel"' >> ${m_conf}
133  echo 'INPUT_DEVICES="evdev synaptics"' >> ${m_conf}
134  mkdir --parents etc/portage/repos.conf
135  cp usr/share/portage/config/repos.conf etc/portage/repos.conf/gentoo.conf
136  cp --dereference /etc/resolv.conf etc/
137 
138  # Chrooting the new environment
139 
140  mount --types proc /proc proc
141  res0=$?
142  mount --rbind /sys sys
143  res1=$?
144  mount --make-rslave sys
145  res2=$?
146  mount --rbind /dev dev
147  res3=$?
148  mount --make-rslave dev
149  res4=$?
150  res=$((${res0} | ${res1} | ${res2} | ${res3} | ${res4}))
151  if test ${res} = 0; then
152  touch stage3
153  else
154  echo "mounting proc exit code: ${res0}" > stage3
155  echo "mounting sys exit code: ${res1}" >> stage3
156  echo "rslave sys exit code: ${res2}" >> stage3
157  echo "mounting dev dev exit code: ${res3}" >> stage3
158  echo "rslave dev exit code exit code: ${res4}" >> stage3
159  echo "Failed to bind liveCD to main disk"
160  sleep 10
161  exit -1
162  fi
163  cd ~
164  chroot /mnt/gentoo /bin/bash mkvm_chroot.sh
165 }

171 finalize() {
172  umount -l /mnt/gentoo/dev{/shm,/pts,}
173  umount /mnt/gentoo/run
