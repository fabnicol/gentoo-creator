37 ## @brief Process options, create Gentoo VirtualBox machine and optionally create clonezilla install medium
38 ## @note This file is not included into the clonezilla ISO liveCD.
39 ## @par USAGE
40 ## @code
41 ## mkgentoo [[switch=argument]...] filename.iso [1]
42 ## mkgentoo [[switch=argument]...] [2]
43 ## mkgentoo help[=md] [3]
44 ## @endcode
45 ## @par
46 ## Usage [1] creates a bootable ISO output file with a current Gentoo distribution. @n
47 ## Usage [2] creates a VirtualBox VDI dynamic disk and a virtual machine with name Gentoo. @n
48 ## Usage [3] prints this help, in markdown form if argument 'md' is specified. @n
49 ## @par
50 ## Run: @code mkgentoo help @endcode to print a list of possible switches and arguments.
51 ## @warning you should have at least 55 GB of free disk space in the current directory or in vmpath
52 ## if specified.
53 ## Boolean values are either 'true' or 'false'. For example, to build a minimal distribution,
54 ## specify <tt> minimal=true</tt> on command line.
55 ## @par \b Examples:
56 ## @li Only create the VM and virtual disk, in debug mode,
57 ## without R or RStudio and set new passwords, for a French-language platform. Use 8 cores.
58 ## @code mkgentoo language=fr minimal=true debug_mode=true ncpus=8 nonroot_user=ken passwd='util!Hx&32F' rootpasswd='Hk_32!_CD' cleanup=false @endcode
59 ## @li Create ISO clonezilla image of Gentoo linux, burn it to DVD, create an installed OS
60 ## on a USB stick whose model label starts with \e PNY and finally create a clonezilla installer
61 ## on another USB stick mounted under <tt> /media/ken/AA45E </tt>
62 ## @code mkgento burn usb_device="PNY" usb_installer="Sams" my_gentoo_image.iso @endcode

137 ## @fn test_cli_pre()
138 ## @brief Check VirtualBox version and prepare commandline analysis
139 ## @retval 0 otherwise exit -1 if VirtualBox is too old
140 ## @ingroup createInstaller
141 
142 test_cli_pre() {
143  declare -r vbox_version=$(VBoxManage -v)
144  declare -r version_major=$(echo ${vbox_version} | sed -E 's/([0-9]+)\..*/\1/')
145  declare -r version_minor=$(echo ${vbox_version} | sed -E 's/[0-9]+\.([0-9]+)\..*/\1/')
146  declare -r version_index=$(echo ${vbox_version} | sed -E 's/[0-9]+\.[0-9]+\.([0-9][0-9]).*/\1/')

152 
153  declare -r -x ISO_OUTPUT=$(echo ${CLI} | sed -E 's/.*(\b\w+)\.(iso|ISO)/\1\.iso/')
154  if test "${ISO_OUTPUT}" != "" -a "${ISO_OUTPUT}" != "${CLI}"; then
155  echo "Build Gentoo distribution to bootable ISO output ${ISO_OUTPUT}"
156  declare -r -x CREATE_ISO="true"
157  else
158  echo "You did not indicate an ISO output file."
159  echo "A Virtual machine will be created with name Gentoo under $HOME"
160  declare -r -x CREATE_ISO="false"
161  fi
162  return 0
163 }

263 ## @fn fetch_livecd()
264 ## @brief Downloads Gentoo install CD
265 ## @details Caches it as ${ISO}
266 ## @retval Returns 0 on success or -1 on exit
267 ## @ingroup createInstaller
268 
269 fetch_livecd() {
270  cd "${VMPATH}"
271  local CACHED_ISO="install-${PROCESSOR}-minimal.iso"
272  if test ${DOWNLOAD} = "false"; then
273  if test -f ${CACHED_ISO}; then
274  cp -vf ${CACHED_ISO} ${ISO}
275  LIVECD=${ISO}
276  return 0
277  else
278  echo "No ISO file was found, please rerun with download=true"
279  exit -1
280  fi
281  fi
282  rm install-${PROCESSOR}-minimal*\.iso*
283  rm latest-install-${PROCESSOR}-minimal*\.txt*
284  local downloaded=""
285  wget ${MIRROR}/releases/${PROCESSOR}/autobuilds/latest-install-${PROCESSOR}-minimal.txt
286  if test $? != 0; then
287  echo "Could not download live CD from Gentoo mirror"
288  exit -1
289  fi
290  local current=$(cat latest-install-${PROCESSOR}-minimal.txt | grep "install-${PROCESSOR}-minimal.*.iso" | sed -E 's/iso.*$/iso/' )
291  local downloaded=$(basename ${current})
292  echo "Downloading $current..."
293  wget "${MIRROR}/releases/${PROCESSOR}/autobuilds/${current}"
294  if test $? != 0; then
295  echo "Could not download live CD"
296  exit -1
297  else
298  if test ${DISABLE_MD5_CHECK} = "false"; then
299  check_md5sum ${downloaded}
300  fi
301  if test -f ${downloaded}; then
302  cp -vf ${downloaded} ${CACHED_ISO}
303  mv ${downloaded} ${ISO}
304  if test -f ${ISO}; then
305  LIVECD=${ISO}
306  else
307  echo "No active ISO (${ISO}) file!"
308  exit -1
309  fi
310  else
311  echo "Could not find downloaded live CD ${downloaded}"
312  exit -1
313  fi
314  fi
315  return 0
316 }

370 ## @fn make_boot_from_livecd()
371 ## @brief Tweak the Gentoo minimal install CD so that the custom-
372 ## made shell scripts and stage3 archive are included into the squashfs filesystem.
373 ## @details This function is returned from early if @code create_squashfs=false @endcode is given on commandline.
374 ## @note Will be run in the ${VM} virtual machine
375 ## @retval Returns 0 on success or -1 on failure.
376 ## @ingroup createInstaller
377 
378 make_boot_from_livecd() {
379  if ! test -f ${ISO}; then
380  echo "No active ISO file in current directory!"
381  exit -1
382  fi
383  if test ${CREATE_SQUASHFS} = "false"; then
384  return 0;
385  fi
386  mountpoint -q mnt && umount -l mnt
387  if test -d mnt; then
388  rm -rf mnt
389  fi
390  mkdir mnt
391  mount -oloop ${ISO} mnt/
392  ! mountpoint -q mnt && echo "ISO not mounted!" && exit -1
393  if test -d mnt2; then
394  rm -rf mnt2
395  fi
396  mkdir mnt2
397  rsync -av mnt/ mnt2
398  cd mnt2/isolinux
399  sed -i 's/timeout.*/timeout 1/' isolinux.cfg
400  sed -i 's/ontimeout.*/ontimeout gentoo/' isolinux.cfg
401  cd ..
402  echo "unsquashfs"
403  unsquashfs image.squashfs
404  if test $? != 0; then
405  echo "unsquashfs failed !"
406  exit -1
407  fi
408  cd ..
409 
410  if ! test -f scripts/mkvm.sh; then
411  echo "No mkvm.sh script!"
412  exit -1
413  fi
414  if ! test -f scripts/mkvm_chroot.sh; then
415  echo "No mkvm_chroot.sh script!"
416  exit -1
417  fi
418  if test "${MINIMAL}" = "true" -a "${ELIST}" = "ebuilds.list"; then
419  cp -vf ${ELIST}.minimal ${ELIST}
420  else
421  cp -vf ${ELIST}.complete ${ELIST}
422  fi
423  if ! test -f ${ELIST}; then
424  echo "No ebuild list!"
425  exit -1
426  fi
427  if ! test -f ${STAGE3}; then
428  echo "No stage3 archive!"
429  exit -1
430  fi
431  if ! test -f ${KERNEL_CONFIG}; then
432  echo "No kernel configuration file!"
433  exit -1
434  fi
435  local sqrt="mnt2/squashfs-root/root/"
436  mv -vf ${STAGE3} ${sqrt}
437  cp -vf scripts/mkvm.sh ${sqrt}
438  chmod +x ${sqrt}mkvm.sh
439  cp -vf scripts/mkvm_chroot.sh ${sqrt}
440  chmod +x ${sqrt}mkvm_chroot.sh
441  cp -vf ${ELIST} ${sqrt}
442  cp -vf ${KERNEL_CONFIG} ${sqrt}
443  cd ${sqrt}
444  rc=".bashrc"
445  cp -vf /etc/bash.bashrc ${rc}
446  declare -i i
447  for ((i=0; i<ARRAY_LENGTH; i++)); do
448  local capname=${ARR[i*3]^^}
449  local expstring="export ${capname}=\"${!capname}\""
450  echo "${expstring}"
451  echo "${expstring}" >> ${rc}
452  done
453  echo "/bin/bash mkvm.sh" >> ${rc}
454  cd ../..
455  rm image.squashfs
456  mksquashfs squashfs-root/ image.squashfs
457  rm -rf squashfs-root/
458  cd ..
459  mkisofs -J -R -o ${ISO} -b isolinux/isolinux.bin -c isolinux/boot.cat -no-emul-boot -boot-load-size 4 -boot-info-table mnt2
460  if test $? != 0; then
461  echo "mkisofs could not recreate the ISO file to boot virtual machine ${VM}"
462  exit -1
463  fi
464  umount -l mnt
465  rm -rvf mnt
466  rm -rvf mnt2
467  return 0
468 }

483 ## @fn delete_vm()
484 ## @param vm VM name
485 ## @param ext virtual disk extension, without dot (defaults to "vdi").
486 ## @brief Powers off, possibly with emergency stop, the VM names as first argument.
487 ## @details @li Unregisters it
488 ## @li Deletes its folder structure and hard drive (default is "vdi" as a second argument)
489 ## @retval Returns 0 if Directory and hard drive could be erased, otherwise the OR value of both
490 ## erasing commands
491 ## @ingroup createInstaller
492 
493 delete_vm() {
494  if test_vm_running "$1"; then
495  VBoxManage controlvm "$1" poweroff
496  fi
497  if test_vm_running "$1"; then
498  VBoxManage startvm $1 --type emergencystop
499  fi
500  if test "$(VBoxManage list vms | grep \"$1\")" != ""; then
501  VBoxManage unregistervm "$1" --delete
502  fi
503 
504  # The following should be unnecessary except for issues with VBoxManage unregistervm
505  # I stubled into such situations a few times
506 
507  if test -d "${VMPATH}/$1"; then
508  rm -rvf "${VMPATH}/$1"
509  fi
510  res=$?
511  if test "$2" != "" -a -f "${VMPATH}/$1.$2"; then
512  rm -f "${VMPATH}/$1.$2"
513  fi
514  sed -i -E "s/^.*${VM}.*$//g" /root/.config/VirtualBox/VirtualBox.xml
515  res=$(($? | ${res}))
516  return ${res}
517 }

519 ## @fn create_vm()
520 ## @brief Create main VirtualBox machine using VBoxManage commandline
521 ## @details Register machine, create VDI drive, create IDE drive attach disks to controlers @n
522 ## Attach augmented clonezilla LiveCD to IDE controller. @n
523 ## Wait for the VM to complete its task. Check that it is still running every minute. @n
524 ## Finally compact it.
525 ## @note VM may be visible (vmtype=gui) or without GUI (vmtype=headless, currently to be fixed)
526 ## @todo Find a way to only compact on success and never on failure of VM.
527 ## @ingroup createInstaller
528 
529 create_vm() {
530  export PATH=${PATH}:${VBPATH}
531  cd ${VMPATH}
532  delete_vm "${VM}" "vdi"
533  VBoxManage createvm --name "${VM}" --ostype gentoo_64 --register --basefolder "${VMPATH}"
534  VBoxManage modifyvm "${VM}" --cpus ${NCPUS} --cpu-profile host --memory ${MEM} --vram 256 --ioapic on --usbxhci on --usbehci on
535  VBoxManage createhd --filename "${VM}.vdi" --size ${SIZE} --variant Standard

579 ## @fn process_clonezilla_iso()
580 ## @brief Download clonezilla ISO or recover it from cache calling #fetch_clonezilla_iso. @n
581 ## Upgrade it with virtualbox guest additions.
582 ## @details Chroot into the clonezilla Ubuntu GNU/Linux distribution and runs apt to build
583 ## kernel modules
584 ## and install the VirtualBox guest additions ISO image. @n
585 ## Upgrade clonezilla kernel consequently
586 ## Recreates the quashfs system after exiting chroot.
587 ## Copy the new \b isolinux.cfg parameter file: automates and silences clonezilla behaviour
588 ## on disk recovery.
589 ## Calls #clonezilla_to_iso
590 ## @note Installing the guest additions is a prerequisite to folder sharing between the ISO VM
591 ## and the host.
592 ## Folder sharing is necessary to recover a compressed clonezilla image of the VDI virtual disk
593 ## into the ISOFILES/home/partimag/image directory.
594 ## @ingroup createInstaller
595 
596 process_clonezilla_iso() {
597  fetch_clonezilla_iso
598  for i in proc sys dev run; do mount -B /$i squashfs-root/$i; done
599  chroot squashfs-root
600  mkdir /boot
601  apt update -yq
602  apt upgrade -yq <<< $(echo N)
603  local headers=$(apt-cache search ^linux-headers | tail -n1 | cut -f 1 -d' ')
604  local kernel=$(apt-cache search ^linux-image | grep -v unsigned | tail -n1 | cut -f 1 -d' ')
605  apt install -qy ${headers}
606  apt install -qy ${kernel}
607  apt install -qy build-essential gcc <<< $(echo N)
608  apt install -qy virtualbox-dkms
609  apt install -qy virtualbox-guest-additions-iso
610  mount -oloop /usr/share/virtualbox/VBoxGuestAdditions.iso /mnt
611  cd /mnt
612  /sbin/rcvboxadd quicksetup all
613  /bin/bash VBoxLinuxAdditions.run
614  cd /
615  umount /mnt
616  apt remove -y -q ${headers} build-essential gcc virtualbox-guest-additions-iso
617  apt autoremove -y -q
618  exit
619  cp -vf --dereference squashfs-root/boot/vmlinuz vmlinuz
620  cp -vf --dereference squashfs-root/boot/initrd.img initrd.img
621  rm -rf squashfs-root/boot
622  rm -vf filesystem.squashfs
623  for i in proc sys dev run; do umount squashfs-root/$i; done
624  mksquashfs squashfs-root filesystem.squashfs
625  rm -rf squashfs-root/
626  cd "${VMPATH}/mnt2"
627  cp -vf ../clonezilla/savedisk/isolinux.cfg syslinux
628  cd "${VMPATH}"
629  rm -vf ${CLONEZILLACD}
630  echo
631  clonezilla_to_iso ${CLONEZILLACD} "mnt2"
632  rm -rf mnt2
