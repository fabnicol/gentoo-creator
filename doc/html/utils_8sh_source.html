29 ## @fn check_md5sum()
30 ## @param filename Local name of file to be checked for md5sum.
31 ## @brief Check md5sums in file MD5SUMS
32 ## @retval Return 0 on success otherwise -1 on exit
33 ## @ingroup auxiliaryFunctions
34 
35 check_md5sum() {
36  local ref=$(cat MD5SUMS | grep "$1" | cut -f 1 -d' ')
37  downloaded=$(md5sum $1 | cut -f 1 -d' ')
38  if test ${downloaded} = ${ref}; then
39  return 0
40  else
41  echo "MD5 checkum for $1 is not correct. Please download manually..."
42  exit -1
43  fi
44 }

54 ## @fn is_block_device()
55 ## @brief Test if argument is non-loop block device
56 ## @param label Label of potential block device (e.g. \b sdc) to be tested.
57 ## @retval Return 0 (true) if input is a block device otherwise 1.
58 ## @ingroup auxiliaryFunctions
59 
60 is_block_device() {
61  local devices=$(list_block_devices)
62  grep -q "$1" <<< "${devices}" && return 0
63  return 1
64 }

66 ## @fn get_mountpoint()
67 ## @brief Gives mount folder from device label input
68 ## @ingroup auxiliaryFunctions
69 
70 get_mountpoint() {
71  if is_block_device "$1"; then
72  echo "$1 is not a block device!"
73  echo "Device labels should be in the following list:"
74  echo $(list_block_devices)
75  exit -1
76  fi
77  echo "$(findmnt --raw --first -a -n -c "$1" | cut -f1 -d' ')"
78 }

80 ## @fn get_device()
81 ## @brief Give device from mount folder input
82 ## @ingroup auxiliaryFunctions
83 
84 get_device() {
85  if test -d "$1"; then
86  device=$(findmnt --raw --first -a -n -c $1 | cut -f2 -d' ')
87  else
88  if is_block_device "$1"; then
89  echo "$1"
90  else
91  echo "$1 is neither a mountpoint nor a block device"
92  exit -1
93  fi
94  fi
95  echo ${device}
96 }

98 ## @fn test_cdrecord()
99 ## @brief Test whether \b cdrecord is functional
100 ## @details Try to scan the SCSI bus and exit on error
101 ## @warning There may be a rights issue with versions of \b cdrecord
102 ## built from original source as they sometimes request elevated
103 ## rights to perform burning with some hardware configurations. @n
104 ## In such an event, either run script with elevated rights or
105 ## modify \b cdrecord rights using: @code
106 ## chown root cdrecord && chgrp bin cdrecord && chmod 04755 cdrecord @endcode
107 ## @retval 0 on success otherwise exits -1 on failure.
108 ## @ingroup auxiliaryFunctions
109 
110 test_cdrecord() {
111  echo "cdrecord scanbus: "
112  ${CDRECORD} -scanbus
113  if test $? != 0; then
114  echo "cdrecord version is not functional"
115  echo "Try reinstalling cdrecord"
116  exit -1
117  fi
118  return 0
119 }

121 ## @fn burn_iso()
122 ## @brief Burn Gentoo clonezilla installer to DVD using \b cdrecord
123 ## @note \b cdrecord should have been installed beforehand
124 ## @retval Return \b cdrecord return code
125 ## @ingroup auxiliaryFunctions
126 
127 burn_iso() {
128  if test "${CDRECORD}" = ""; then
129  if $(which cdrecord); then
130  echo "Could not find cdrecord"
131  echo "Please install cdrtools in your PATH or specify cdrecord full filepath on commandline:"
132  echo "burn=true cdrecord=/path/to/cdrecord/executable"
133  return -1
134  else
135  CDRECORD=$(which cdrecord)
136  test_cdrecord
137  fi
138  else
139  test_cdrecord
140  fi
141  echo "Burning installation medium to optical disc..."
142  if test "${SCSI_ADDRESS}" = ""; then
143  ${CDRECORD} -eject "${ISO_OUTPUT}"
144  else
145  ${CDRECORD} -eject dev=${SCSI_ADDRESS} "${ISO_OUTPUT}"
146  fi
147 }

154 create_install_usb_device() {
