43 ## @fn check_md5sum()
44 ## @param filename Local name of file to be checked for md5sum.
45 ## @brief Check md5sums in file MD5SUMS
46 ## @retval Return 0 on success otherwise -1 on exit
47 ## @ingroup auxiliaryFunctions
48 
49 check_md5sum() {
50  local ref=$(cat MD5SUMS | grep "$1" | cut -f 1 -d' ')
51  downloaded=$(md5sum $1 | cut -f 1 -d' ')
52  [ ${downloaded} = ${ref} ] && return 0
53  logger -s "[ERR] MD5 checkum for $1 is not correct. \
54 Please download manually..."
55  exit -1
56 }

99 ## @brief Test if argument is non-loop block device
100 ## @param label Label of potential block device (e.g. \b sdc) to be

143 ## @fn test_cdrecord()
144 ## @brief Test whether \b cdrecord is functional
145 ## @details Try to scan the SCSI bus and exit on error
146 ## @warning There may be a rights issue with versions of \b cdrecord
147 ## built from original source as they sometimes request elevated
148 ## rights to perform burning with some hardware configurations. @n
149 ## In such an event, either run script with elevated rights or
150 ## modify \b cdrecord rights using: @code
151 ## chown root cdrecord && chgrp bin cdrecord && chmod 04755 cdrecord
152 ## @endcode
153 ## @retval 0 on success otherwise exits -1 on failure.
154 ## @ingroup auxiliaryFunctions
155 
156 test_cdrecord() {
157  logger -s "[MSG] cdrecord scanbus: "
158  ${CDRECORD} -scanbus
159  [ $? != 0 ] \
160  && { logger -s "[ERR] cdrecord version is not functional"
161  logger -s "[MSG] Try reinstalling cdrecord"
162  exit -1; }
163  return 0
164 }

166 ## @fn recreate_liveCD_ISO()
167 ## @brief Create ISO of liveCD out of directory
168 ## @param dir Directory containing all files
169 ## @return 0 on success or exits -1 on failure.
170 ## @note An alternative xorriso solution could be considered
171 
172 recreate_liveCD_ISO() {
173 
174  "${VERBOSE}" \
175  && mkisofs -v -J -R -o "${ISO}" \
176  -b ${ISOLINUX_DIR}/isolinux.bin \
177  -c ${ISOLINUX_DIR}/boot.cat -no-emul-boot \
178  -boot-load-size 4 \
179  -boot-info-table "$1" \
180  || mkisofs -J -R -o "${ISO}" \
181  -b ${ISOLINUX_DIR}/isolinux.bin \
182  -c ${ISOLINUX_DIR}/boot.cat -no-emul-boot \
183  -boot-load-size 4 \
184  -boot-info-table "$1" 2>/dev/null 1>/dev/null
185 
186 # mkisofs almost never fails but if it does, hard stop here.
187 
188  [ $? != 0 ] && {
189  logger -s "[ERR] mkisofs could not recreate the ISO file to \
190 boot virtual machine ${VM} from directory $1"
191  exit -1; }
192 }

194 ## @fn burn_iso()
195 ## @brief Burn Gentoo clonezilla installer to DVD using \b cdrecord
196 ## @note \b cdrecord should have been installed beforehand
197 ## @retval Return \b cdrecord return code
198 ## @ingroup auxiliaryFunctions
199 
200 burn_iso() {
201  if [ -z "${CDRECORD}" ]
202  then
203  if $(which cdrecord)
204  then
205  logger -s "[ERR] Could not find cdrecord"
206  logger -s "[ERR] Please install cdrtools in your PATH or \
207  specify cdrecord full filepath on commandline:"
208  logger -s " burn=true \
209 cdrecord=/path/to/cdrecord/executable"
210  return -1
211  else
212  CDRECORD=$(which cdrecord)
213  test_cdrecord
214  fi
215  else
216  test_cdrecord
217  fi
218  logger -s "[INF] Burning installation medium to optical disc..."
219  [ -z "${SCSI_ADDRESS}" ] && ${CDRECORD} -eject "${ISO_OUTPUT}" \
220  || ${CDRECORD} -eject dev=${SCSI_ADDRESS} "${ISO_OUTPUT}"
221 }
