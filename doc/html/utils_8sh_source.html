29 ## @fn check_md5sum()
30 ## @param filename Local name of file to be checked for md5sum.
31 ## @brief Check md5sums in file MD5SUMS
32 ## @retval Return 0 on success otherwise -1 on exit
33 ## @ingroup auxiliaryFunctions
34 
35 check_md5sum() {
36  local ref=$(cat MD5SUMS | grep "$1" | cut -f 1 -d' ')
37  downloaded=$(md5sum $1 | cut -f 1 -d' ')
38  [ ${downloaded} = ${ref} ] && return 0
39  || { logger -s "MD5 checkum for $1 is not correct. Please download manually..."; exit -1; }
40 }

50 ## @fn is_block_device()
51 ## @brief Test if argument is non-loop block device
52 ## @param label Label of potential block device (e.g. \b sdc) to be tested.
53 ## @retval Return 0 (true) if input is a block device otherwise 1.
54 ## @ingroup auxiliaryFunctions
55 
56 is_block_device() {
57  local devices=$(list_block_devices)
58  grep -q "$1" <<< "${devices}" && return 0
59  return 1
60 }

63 ## @brief Gives mount folder from device label input
64 ## @ingroup auxiliaryFunctions

66 get_mountpoint() {

68  then
69  logger -s "$1 is not a block device!"

81 get_device() {

84  device=$(findmnt --raw --first -a -n -c $1 | cut -f2 -d' ')
85  echo ${device}
86  else

92 ## @fn test_cdrecord()
93 ## @brief Test whether \b cdrecord is functional
94 ## @details Try to scan the SCSI bus and exit on error
95 ## @warning There may be a rights issue with versions of \b cdrecord
96 ## built from original source as they sometimes request elevated
97 ## rights to perform burning with some hardware configurations. @n
98 ## In such an event, either run script with elevated rights or
99 ## modify \b cdrecord rights using: @code
100 ## chown root cdrecord && chgrp bin cdrecord && chmod 04755 cdrecord @endcode
101 ## @retval 0 on success otherwise exits -1 on failure.
102 ## @ingroup auxiliaryFunctions
103 
104 test_cdrecord() {
105  logger -s "cdrecord scanbus: "
106  ${CDRECORD} -scanbus
107  if [ $? != 0 ]
108  then
109  logger -s "cdrecord version is not functional"
110  logger -s "Try reinstalling cdrecord"
111  exit -1
112  fi
113  return 0
114 }

116 ## @fn recreate_liveCD_ISO
117 ## @brief Create ISO of liveCD out of directory
118 ## @param dir Directory containing all files
119 ## @return 0 on success or exits -1 on failure.
120 ## @note An alternative xorriso solution could be considered
121 
122 recreate_liveCD_ISO() {
123 # using mkisofs out of convention but xorriso could do as well
124 
125  mkisofs ${verb2} -J -R -o ${ISO} \
126  -b ${ISOLINUX_DIR}/isolinux.bin \
127  -c ${ISOLINUX_DIR}/boot.cat -no-emul-boot \
128  -boot-load-size 4 \
129  -boot-info-table "$1"
130 
131 # mkisofs almost never fails but if it does, hard stop here.
132 
133 if [ $? != 0 ]; then
134  logger -s "mkisofs could not recreate the ISO file to boot virtual machine ${VM} from directory $1"
135  exit -1
136 fi
137 }

139 ## @fn burn_iso()
140 ## @brief Burn Gentoo clonezilla installer to DVD using \b cdrecord
141 ## @note \b cdrecord should have been installed beforehand
142 ## @retval Return \b cdrecord return code
143 ## @ingroup auxiliaryFunctions
144 
145 burn_iso() {
146  if [ -z "${CDRECORD}" ]
147  then
148  if $(which cdrecord)
149  then
150  logger -s "Could not find cdrecord"
151  logger -s "Please install cdrtools in your PATH or specify cdrecord full filepath on commandline:"
152  logger -s "burn=true cdrecord=/path/to/cdrecord/executable"
153  return -1
154  else
155  CDRECORD=$(which cdrecord)
156  test_cdrecord
157  fi
158  else
159  test_cdrecord
160  fi
161  logger -s "Burning installation medium to optical disc..."
162  if [ -z "${SCSI_ADDRESS}" ]
163  then
164  ${CDRECORD} -eject "${ISO_OUTPUT}"
165  else
166  ${CDRECORD} -eject dev=${SCSI_ADDRESS} "${ISO_OUTPUT}"
167  fi
168 }
